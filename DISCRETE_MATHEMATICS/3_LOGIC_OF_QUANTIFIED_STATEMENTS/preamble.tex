% Uncomment this to have the outline at the beginning of each section highlighted.
\AtBeginSection[]{
  \begin{frame}{Topic}
    \tableofcontents[
      currentsection,
      hideallsubsections
    ]
  \end{frame}
}

\AtBeginSubsection[]{
  \begin{frame}{Subtopic}
    \tableofcontents[
      sections=\thesection,
      currentsubsection
    ]
  \end{frame}
}
